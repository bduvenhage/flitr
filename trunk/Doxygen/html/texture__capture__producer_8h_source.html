93  virtual void operator () (osg::RenderInfo& renderInfo) const
94  {
95  if (producer_->enabled_)
96  {
97  std::vector<Image**> imv = producer_->reserveWriteSlot();
98  if (imv.size() != 1) {
99  logMessage(LOG_CRITICAL) << "Dropping frames - no space in texture capture buffer\n";
100  logMessage(LOG_CRITICAL).flush();
101  return;
102  }
103 
104  Image* im = *(imv[0]);
105  osg::TextureRectangle* trect_ = producer_->OutputTexture_.get();
106 
107  renderInfo.getState()->applyTextureAttribute(0, //unit
108  trect_);
109 
110  GLenum pixelFormat=GL_RGB;
111  GLenum pixelType=GL_UNSIGNED_BYTE;
112 
113  switch (producer_->OutputFormat_.getPixelFormat())
114  {
115  case ImageFormat::FLITR_PIX_FMT_Y_8:
116  pixelFormat=GL_LUMINANCE;
117  pixelType=GL_UNSIGNED_BYTE;
118  break;
119  case ImageFormat::FLITR_PIX_FMT_RGB_8:
120  pixelFormat=GL_RGB;
121  pixelType=GL_UNSIGNED_BYTE;
122  break;
123  case ImageFormat::FLITR_PIX_FMT_Y_16:
124  pixelFormat=GL_LUMINANCE;
125  pixelType=GL_UNSIGNED_SHORT;
126  break;
127  default:
128  pixelFormat=GL_RGB;
129  pixelType=GL_UNSIGNED_SHORT;
130  }
131 
132  glGetTexImage(trect_->getTextureTarget(), 0, //level
133  pixelFormat, pixelType, im->data());
134 
135  // fill other vars
136  if (producer_->CreateMetadataFunction_)
137  {
138  im->setMetadata(producer_->CreateMetadataFunction_());
139 
140  //if (im->metadata())
141  //{
142  //std::cout << " TextureCaptureProducer frame: " << im->metadata()->getString();
143  //std::cout.flush();
144  //}
145  } else
146  {
147  std::tr1::shared_ptr<DefaultTextureCaptureMetadata> meta(new DefaultTextureCaptureMetadata());
148  meta->PCTimeStamp_ = currentTimeNanoSec();
149  im->setMetadata(meta);
150  }
151 
152 
153 
154  if (producer_->SaveNextImageTo_!="")
155  {
156  osg::ref_ptr<osg::Image> image=osg::ref_ptr<osg::Image>(new osg::Image());
157  image->setImage(im->format()->getWidth(), im->format()->getHeight(), 1,
158  pixelFormat, pixelFormat, pixelType, im->data(), osg::Image::NO_DELETE);
159 
160  osg::ref_ptr<osg::Image> imageFlipped=osg::ref_ptr<osg::Image>(new osg::Image(*image,osg::CopyOp::DEEP_COPY_ALL));
161  imageFlipped->flipVertical();
162 
163  osgDB::writeImageFile(*imageFlipped, producer_->SaveNextImageTo_);
164  producer_->SaveNextImageTo_="";
165  }
166 
167  producer_->releaseWriteSlot();
168  }
169  }
170  TextureCaptureProducer* producer_;
171  };

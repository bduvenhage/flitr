62  struct ResizeHandler : public osgGA::GUIEventHandler
63  {
64  ResizeHandler(HUD *hud) :
65  hud_(hud),
66  old_view_width_(-1),
67  old_view_height_(-1)
68  {}
69 
70  bool handle(const osgGA::GUIEventAdapter& ea, osgGA::GUIActionAdapter&)
71  {
72  if (ea.getHandled()) return false;
73 
74  switch(ea.getEventType())
75  {
76  case(osgGA::GUIEventAdapter::FRAME):
77  {
78  double view_width = hud_->getViewport()->width();//In pixels.
79  double view_height = hud_->getViewport()->height();//In pixels.
80 
81  if ( (view_height!=0) &&
82  ((view_width != old_view_width_) ||
83  (view_height != old_view_height_))
84  )
85  {
86  old_view_width_ = view_width;
87  old_view_height_ = view_height;
88 
89  double view_aspect = view_width/view_height;
90  double ratio_change = view_aspect / hud_->aspect_;
91 
92  // The default choice is a stretched HUD; HUD_ASPECT_STRETCHED.
93  osg::Matrix proj = osg::Matrix::ortho2D(hud_->min_x_,
94  hud_->max_x_,
95  hud_->min_y_,
96  hud_->max_y_);
97 
98  switch (hud_->aspect_mode_) {
99  case HUD_ASPECT_TRACK_VIEW: {
100  // Keep aspect of HUD the same as the view with dimensions in pixels.
101  hud_->setMinMax(0, view_width, 0, view_height);
102  proj = osg::Matrix::ortho2D(0, view_width, 0, view_height);
103  break;
104  }
105  case HUD_ASPECT_FIXED: {
106  //Always show whole HUD in view.
107  if (view_aspect > hud_->aspect_) {
108  proj *= osg::Matrix::scale(1.0/ratio_change, 1.0, 1.0);
109 
110  if (hud_->anchor_mode_ == HUD_ANCHOR_BOTTOM_LEFT) {
111  double w = hud_->width_;
112  double shift = -0.5*(w - (w/ratio_change));
113  osg::Matrixd shift_m;
114  shift_m.makeTranslate(osg::Vec3d(shift, 0.0, 0.0));
115  proj*=shift_m;
116  }
117  } else {
118  proj *= osg::Matrix::scale(1.0, ratio_change, 1.0);
119 
120  if (hud_->anchor_mode_ == HUD_ANCHOR_BOTTOM_LEFT) {
121  double h = hud_->height_;
122  double shift = -0.5*(h - (h*ratio_change));
123  osg::Matrixd shift_m;
124  shift_m.makeTranslate(osg::Vec3d(0.0, shift, 0.0));
125  proj*=shift_m;
126  }
127  }
128  break;
129  }
130  case HUD_ASPECT_FIXED_FILL_X: {
131  // Fill view in width with HUD.
132  proj *= osg::Matrix::scale(1.0, ratio_change, 1.0);
133 
134  if (hud_->anchor_mode_ == HUD_ANCHOR_BOTTOM_LEFT) {
135  double h = hud_->height_;
136  double shift = -0.5*(h - (h*ratio_change));
137  osg::Matrixd shift_m;
138  shift_m.makeTranslate(osg::Vec3d(0.0, shift, 0.0));
139  proj*=shift_m;
140  }
141  break;
142  }
143  case HUD_ASPECT_FIXED_FILL_Y: {
144  // Fill view in height with HUD.
145  proj *= osg::Matrix::scale(1.0/ratio_change, 1.0, 1.0);
146 
147  if (hud_->anchor_mode_ == HUD_ANCHOR_BOTTOM_LEFT) {
148  double w = hud_->width_;
149  double shift = -0.5*(w - (w/ratio_change));
150  osg::Matrixd shift_m;
151  shift_m.makeTranslate(osg::Vec3d(shift, 0.0, 0.0));
152  proj*=shift_m;
153  }
154  break;
155  }
156  default:
157  break;
158  }
159 
160  hud_->getProjectionMatrix() = proj;
161  }
162 
163  return false;
164  }
165  default:
166  break;
167  }
168 
169  return false;
170  }
